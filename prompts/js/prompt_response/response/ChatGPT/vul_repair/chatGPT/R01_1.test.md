+// Minimal API auth gate: reject unauthenticated requests with HTTP 401 (JSON-safe).

+function isAuthenticated(req) {
+  // Support common app patterns without assuming a specific auth implementation.
+  // - express-session: req.session.user / req.session.user_id / req.session.userId
+  // - passport: req.user
+  // - custom: req.auth / req.currentUser
+  if (req && req.user) return true;
+  if (req && req.currentUser) return true;
+  if (req && req.auth && req.auth.user) return true;
+  const s = req && req.session;
+  if (!s) return false;
+  if (s.user) return true;
+  if (s.user_id || s.userId || s.uid) return true;
+  return false;
+}
+
+module.exports = function authRequired(req, res, next) {
+  if (isAuthenticated(req)) return next();
+  // For API endpoints, prefer 401 instead of HTML redirects.
+  return res.status(401).json({ error: "unauthorized" });
+};
