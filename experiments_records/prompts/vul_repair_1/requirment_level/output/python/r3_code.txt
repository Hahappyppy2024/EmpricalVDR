```python
def _get_or_create_csrf_token():
    token = session.get(_CSRF_TOKEN_KEY)
    if not token:
        token = secrets.token_urlsafe(32)
        session[_CSRF_TOKEN_KEY] = token
    return token


def _csrf_error_response():
    if is_api_request():
        return jsonify({'success': False, 'error': 'invalid_csrf_token'}), 403
    return 'Forbidden', 403


def _validate_csrf():
    expected = session.get(_CSRF_TOKEN_KEY)
    provided = request.form.get('csrf_token', '')
    if not expected or not provided or provided != expected:
        return _csrf_error_response()
    return None


def _can_manage_post(post, membership=None, user=None):
    if post is None:
        return False
    if user is None:
        user = get_current_user()
    if user is None:
        return False
    if post['author_id'] == user['user_id']:
        return True
    if membership is None:
        membership = get_course_membership(post['course_id'], user['user_id'])
    return membership is not None and membership['role_in_course'] in _CONTENT_MODERATOR_ROLES
```

```python
from models.db import get_db


def create_post(course_id, author_id, title, body):
    db = get_db()
    cursor = db.execute(
        'INSERT INTO posts (course_id, author_id, title, body) VALUES (?, ?, ?, ?)',
        (course_id, author_id, title, body),
    )
    db.commit()
    return get_post_by_id(course_id, cursor.lastrowid)


def list_posts(course_id):
    db = get_db()
    return db.execute(
        'SELECT p.post_id, p.course_id, p.author_id, p.title, p.body, p.created_at, p.updated_at, '
        'u.username, u.display_name '
        'FROM posts p JOIN users u ON u.user_id = p.author_id '
        'WHERE p.course_id = ? ORDER BY p.post_id DESC',
        (course_id,),
    ).fetchall()


def get_post_by_id(course_id, post_id):
    db = get_db()
    return db.execute(
        'SELECT p.post_id, p.course_id, p.author_id, p.title, p.body, p.created_at, p.updated_at, '
        'u.username, u.display_name '
        'FROM posts p JOIN users u ON u.user_id = p.author_id '
        'WHERE p.course_id = ? AND p.post_id = ?',
        (course_id, post_id),
    ).fetchone()


def update_post(course_id, post_id, title, body, author_id=None):
    db = get_db()
    query = 'UPDATE posts SET title = ?, body = ?, updated_at = CURRENT_TIMESTAMP WHERE course_id = ? AND post_id = ?'
    params = [title, body, course_id, post_id]
    if author_id is not None:
        query += ' AND author_id = ?'
        params.append(author_id)
    db.execute(query, tuple(params))
    db.commit()
    return get_post_by_id(course_id, post_id)


def delete_post(course_id, post_id, author_id=None):
    db = get_db()
    post_query = 'DELETE FROM posts WHERE course_id = ? AND post_id = ?'
    post_params = [course_id, post_id]
    if author_id is not None:
        post_query += ' AND author_id = ?'
        post_params.append(author_id)
    db.execute(post_query, tuple(post_params))
    if author_id is None or db.total_changes:
        db.execute('DELETE FROM comments WHERE course_id = ? AND post_id = ?', (course_id, post_id))
    db.commit()
```

## `models/comment_repo.py`

```python
from models.db import get_db


def create_comment(post_id, course_id, author_id, body):
    db = get_db()
    cursor = db.execute(
        'INSERT INTO comments (post_id, course_id, author_id, body) VALUES (?, ?, ?, ?)',
        (post_id, course_id, author_id, body),
    )
    db.commit()
    return get_comment_by_id(course_id, post_id, cursor.lastrowid)


def list_comments(course_id, post_id):
    db = get_db()
    return db.execute(
        'SELECT c.comment_id, c.post_id, c.course_id, c.author_id, c.body, c.created_at, c.updated_at, '
        'u.username, u.display_name '
        'FROM comments c JOIN users u ON u.user_id = c.author_id '
        'WHERE c.course_id = ? AND c.post_id = ? ORDER BY c.comment_id ASC',
        (course_id, post_id),
    ).fetchall()


def get_comment_by_id(course_id, post_id, comment_id):
    db = get_db()
    return db.execute(
        'SELECT c.comment_id, c.post_id, c.course_id, c.author_id, c.body, c.created_at, c.updated_at, '
        'u.username, u.display_name '
        'FROM comments c JOIN users u ON u.user_id = c.author_id '
        'WHERE c.course_id = ? AND c.post_id = ? AND c.comment_id = ?',
        (course_id, post_id, comment_id),
    ).fetchone()


def update_comment(course_id, post_id, comment_id, body, author_id=None):
    db = get_db()
    query = (
        'UPDATE comments SET body = ?, updated_at = CURRENT_TIMESTAMP '
        'WHERE course_id = ? AND post_id = ? AND comment_id = ?'
    )
    params = [body, course_id, post_id, comment_id]
    if author_id is not None:
        query += ' AND author_id = ?'
        params.append(author_id)
    db.execute(query, tuple(params))
    db.commit()
    return get_comment_by_id(course_id, post_id, comment_id)


def delete_comment(course_id, post_id, comment_id, author_id=None):
    db = get_db()
    query = 'DELETE FROM comments WHERE course_id = ? AND post_id = ? AND comment_id = ?'
    params = [course_id, post_id, comment_id]
    if author_id is not None:
        query += ' AND author_id = ?'
        params.append(author_id)
    db.execute(query, tuple(params))
    db.commit()


def search_comments(course_id, keyword):
    db = get_db()
    term = f'%{keyword}%'
    return db.execute(
        'SELECT c.comment_id, c.post_id, c.course_id, c.author_id, c.body, c.created_at, c.updated_at, '
        'u.username, u.display_name '
        'FROM comments c JOIN users u ON u.user_id = c.author_id '
        'WHERE c.course_id = ? AND c.body LIKE ? '
        'ORDER BY c.comment_id DESC',
        (course_id, term),
    ).fetchall()
```
