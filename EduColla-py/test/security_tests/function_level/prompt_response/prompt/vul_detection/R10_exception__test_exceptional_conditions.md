def _no_leak(text):
    forbidden = [
        r"Traceback", r"Werkzeug", r"Debugger",
        r"File \"[^\"]+\.py\"", r"/[A-Za-z0-9_\-./]+\.py", r"\\[A-Za-z0-9_\-\\.]+\.py",

    for pat in forbidden:
